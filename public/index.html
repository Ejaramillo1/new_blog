db <- data %>%
  select(id, per_ocu, nombre_act, entidad, municipio, ageb, latitud, longitud, fecha_alta, codigo_act) %>%
  mutate(code = str_extract(string = as.character(codigo_act), pattern = "(\\d\\d)"),
